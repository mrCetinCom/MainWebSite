    <script>
        document.getElementById('current-year').textContent = new Date().getFullYear();

        const translations = {
            tr: {
                title: "Geliştiriliyor!",
                message: "Dijital dönüşümün merkezinde, güçlü bir vizyon ve sağlam bir mühendislik anlayışıyla yola çıktık. Hayatı kolaylaştıran, güven veren yazılım çözümleriyle çok yakında burada olacağız. Sizi tanımak ve birlikte değer üretmek için sabırsızlanıyoruz.",

                contact: "Gelişmelerden haberdar olmak için:",
                footer: `&copy; ${new Date().getFullYear()} mrCetin. Tüm Hakları Saklıdır.`,
                companyLogo: "Şirket Logosu"
            },
            en: {
                title: "Under Development!",
                message: "At the heart of digital transformation, we’re moving forward with a strong vision and a solid engineering mindset. Very soon, we’ll be here — with software solutions that simplify life and inspire trust. We’re excited to connect and create meaningful value together.",
                loading: "LOADING...",
                contact: "To be informed about the developments:",
                footer: `&copy; ${new Date().getFullYear()} mrCetin. All Rights Reserved.`,
                companyLogo: "Company Logo"
            }
        };

        function setLanguage(lang) {
            document.documentElement.lang = lang;
            const elements = document.querySelectorAll('[data-lang]');
            elements.forEach(element => {
                const key = element.getAttribute('data-lang');
                if (translations[lang] && translations[lang][key]) {
                    element.innerHTML = translations[lang][key];
                }
            });

            const altElements = document.querySelectorAll('[data-lang-alt]');
            altElements.forEach(element => {
                const key = element.getAttribute('data-lang-alt');
                if (translations[lang] && translations[lang][key]) {
                    element.alt = translations[lang][key];
                }
            });

            document.querySelectorAll('.lang-switcher button').forEach(button => {
                button.classList.remove('active');
            });
            document.querySelector(`.lang-switcher button[onclick="setLanguage('${lang}')"]`).classList.add('active');
        }

        // Set default language
        setLanguage('tr');
    </script>
